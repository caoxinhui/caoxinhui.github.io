### 判断两个字符串是否为变形词
```js
function isDeformation(str1, str2) {
  if (!str1 || !str2 || str1.length !== str2.length) {
    return false
  }
  let map = Array(256).fill(0)
  for (let i = 0; i < str1.length; i++) {
    map[str1[i]]++
  }
  for (let i = 0; i < str2.length; i++) {
    if (map[str2[i]]-- === 0) {
      return false
    }
  }
  return true
}
```
