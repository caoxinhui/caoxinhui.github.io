function process(arr, i, from, mid, to) {
  if (i === -1) {
    return 0
  }
  if (arr[i] !== from && arr[i] !== to) return -1
  if (arr[i] === from) {
    return process(arr, i - 1, from, to, mid)
  } else {
    let rest = process(arr, i - 1, mid, from, to)
    if (rest === -1) return -1
    return 1 << i + rest
  }
